    <script>
        let allData = [];

        document.getElementById("upload-form").addEventListener("submit", function (event) {
            event.preventDefault();
            let formData = new FormData(this);
            let resultDiv = document.getElementById("result");
            let resultTable = document.getElementById("result-table");
            let downloadLink = document.getElementById("download-link");

            resultDiv.style.display = "block";
            resultTable.innerHTML = "";

            fetch("/upload", { method: "POST", body: formData })
                .then(response => response.json())
                .then(data => {
                    if (data.error) {
                        alert(data.error);
                    } else {
                        allData = data.results;
                        updateTable(allData);
                        downloadLink.href = data.fileUrl;
                        downloadLink.classList.remove("d-none");
                    }
                });
        });

        function updateTable(data) {
            let resultTable = document.getElementById("result-table");
            resultTable.innerHTML = "";
            let totalcommission = 0 , totalshipping = 0 , totalSupply = 0, totalcollection = 0, totalProfit = 0, totalcollect = 0, totalfuel = 0;

            data.forEach((row, index) => {
                let tr = document.createElement("tr");
                tr.innerHTML = `<td>${index + 1}</td> <!-- رقم الصف -->
                        <td>${row.waybill}</td>
                        <td>${row.sender}</td>
                        <td>${row.weight}</td>
                        <td>${row.shipping}</td>
                        <td>${row.commission}</td>
                        <td>${row.total}</td>
                        <td>${row.profit}</td>
                        <td>${row.collection}</td>
                        <td>${row.supply}</td>
                        <td style="background-color: ${row.typeColor}">${row.type}</td>`;
                resultTable.appendChild(tr);

                totalcollect += row.supply;


                totalshipping +=   row.shipping;
                totalcommission +=   row.commission;

                totalProfit = totalshipping - totalcommission -0.1*totalcommission   ;

                totalcollection += row.collection;

                totalfuel += row.fuel;
                totalSupply = totalcollect    ;

            });

            document.getElementById("total-supply").textContent = totalSupply.toFixed(2);
            document.getElementById("total-profit").textContent = totalProfit.toFixed(2);
            document.getElementById("total-collection").textContent = totalcollection.toFixed(2);
        }

        document.getElementById("search-box").addEventListener("input", function () {
            let query = this.value.toLowerCase();
            let suggestionsDiv = document.getElementById("suggestions");
            suggestionsDiv.innerHTML = "";

            if (query.length > 0) {
                let filteredNames = [...new Set(allData.map(row => row.sender))]
                    .filter(name => name.toLowerCase().includes(query));

                filteredNames.forEach(name => {
                    let suggestion = document.createElement("a");
                    suggestion.classList.add("list-group-item", "list-group-item-action");
                    suggestion.textContent = name;
                    suggestion.addEventListener("click", function () {
                        document.getElementById("search-box").value = name;
                        filterByName(name);
                        suggestionsDiv.innerHTML = "";
                    });
                    suggestionsDiv.appendChild(suggestion);
                });
            }
        });
        let filteredData = [];
        function filterByName(name) {
    filteredData = allData.filter(row => row.sender === name);  
    updateTable(filteredData);
}
